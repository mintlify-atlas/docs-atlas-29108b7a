## Overview

The Anime Theme resource represents opening (OP), ending (ED), and insert (IN) songs for anime. Each theme is associated with an anime, song, and may have a group.

## List Anime Themes

<ParamField path="GET" type="endpoint">
  /api/animetheme
</ParamField>

Retrieve a paginated list of anime themes.

### Query Parameters

<ParamField query="page[size]" type="integer">
  Number of results per page (default: 15, max: 100)
</ParamField>

<ParamField query="page[number]" type="integer">
  Page number to retrieve
</ParamField>

<ParamField query="include" type="string">
  Comma-separated list of relationships to include. Available includes:
  - `anime` - Parent anime
  - `anime.images` - Anime images
  - `song` - Associated song
  - `song.artists` - Song artists
  - `song.performances` - Song performances
  - `song.performances.artist` - Performance artists
  - `group` - Theme group (if applicable)
  - `animethemeentries` - Theme entries
  - `animethemeentries.videos` - Videos for each entry
  - `animethemeentries.videos.audio` - Audio tracks
</ParamField>

<ParamField query="filter[type]" type="string">
  Filter by theme type. Values: `OP`, `ED`, `IN`
</ParamField>

<ParamField query="filter[sequence]" type="integer">
  Filter by sequence number
</ParamField>

<ParamField query="filter[anime_id]" type="integer">
  Filter by anime ID
</ParamField>

<ParamField query="filter[song_id]" type="integer">
  Filter by song ID
</ParamField>

<ParamField query="fields[animetheme]" type="string">
  Comma-separated list of fields to return. Available fields:
  - `theme_id`
  - `anime_id`
  - `song_id`
  - `group_id`
  - `type`
  - `sequence`
  - `slug`
</ParamField>

<ParamField query="sort" type="string">
  Sort by field. Prefix with `-` for descending order.
  Available: `type`, `sequence`, `created_at`, `updated_at`
</ParamField>

<ParamField query="q" type="string">
  Full-text search query across themes, anime names, and song titles
</ParamField>

### Response

<ResponseField name="animethemes" type="array">
  Array of anime theme objects
  
  <ResponseField name="theme_id" type="integer">
    Unique identifier
  </ResponseField>
  
  <ResponseField name="anime_id" type="integer">
    Parent anime ID
  </ResponseField>
  
  <ResponseField name="song_id" type="integer" optional>
    Associated song ID
  </ResponseField>
  
  <ResponseField name="group_id" type="integer" optional>
    Theme group ID
  </ResponseField>
  
  <ResponseField name="type" type="string">
    Theme type (OP, ED, IN)
  </ResponseField>
  
  <ResponseField name="sequence" type="integer" optional>
    Sequence number for multiple OPs/EDs
  </ResponseField>
  
  <ResponseField name="slug" type="string">
    Generated slug (e.g., "OP1", "ED2")
  </ResponseField>
  
  <ResponseField name="created_at" type="string">
    ISO 8601 timestamp
  </ResponseField>
  
  <ResponseField name="updated_at" type="string">
    ISO 8601 timestamp
  </ResponseField>
</ResponseField>

</ParamField>

</ParamField>
